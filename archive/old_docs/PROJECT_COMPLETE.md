tree = interface.create_task_tree("Social Media Platform")

auth = tree.add_task("Authentication", agent="mira", depends_on=[backend])
api = tree.add_task("REST API", agent="nova", depends_on=[backend])
db = tree.add_task("Database", agent="ivy", depends_on=[backend])

# Frontend
frontend = tree.add_task("React App", agent="felix")
ui = tree.add_task("UI Components", agent="felix", depends_on=[frontend])

# Integration
integration = tree.add_task("Frontend-Backend", agent="link", 
                            depends_on=[api, ui])

# Quality
tests = tree.add_task("Test Suite", agent="quinn", 
                     depends_on=[integration])
security = tree.add_task("Security Audit", agent="mira", 
                        depends_on=[integration])
review = tree.add_task("Code Review", agent="orion", 
                      depends_on=[tests, security])

# Execute with parallel processing
result = await interface.execute_task_tree(tree, parallel=True)
